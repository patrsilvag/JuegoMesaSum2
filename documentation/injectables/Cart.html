/**
 * @description Servicio de carrito. Gestiona productos, cantidades

  private descuento = 0;

  // ðŸ”¹ Aplicar descuento (llamado desde el formulario reactivo)

   * @param p Producto a agregar.

  // ðŸ”¹ Agregar al carrito
  agregar(p: CartItem) {
    const actual = [...this.items.value];
    const existe = actual.find((i) => i.id === p.id);

    if (existe) {
      existe.cantidad += p.cantidad;
    } else {
      actual.push({ ...p });
    }

    this.items.next(actual);
  }

  /**
   * @description Incrementa en 1 la cantidad de un Ã­tem del carrito.
   * @param id Identificador del producto.

  // ðŸ”¹ Sumar cantidad

    const actual = [...this.items.value];
    const item = actual.find((i) => i.id === id);

    if (item) {
      item.cantidad++;
      this.items.next(actual);
    }
  }

  /**
   * @description Decrementa en 1 la cantidad de un Ã­tem del carrito.
   * Elimina el Ã­tem si la cantidad llega a 0.
   * @param id Identificador del producto.

  // ðŸ”¹ Restar cantidad
  restar(id: string) {
    const actual = [...this.items.value];
    const item = actual.find((i) => i.id === id);

    if (item && item.cantidad > 1) {
      item.cantidad--;
      this.items.next(actual);
    }
  }

  /**
   * @description Elimina por completo un producto del carrito.
   * @param id Identificador del producto a quitar.

  // ðŸ”¹ Eliminar producto
  quitarProducto(id: string) {
    const actual = this.items.value.filter((i) => i.id !== id);
    this.items.next(actual);
  }

  total() {

    return base - (base * this.descuento) / 100;
  }
