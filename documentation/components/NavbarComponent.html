import { Component, OnInit, OnDestroy } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterModule } from '@angular/router';
import { Subscription } from 'rxjs'; // Necesaria para manejar la desuscripci贸n

import { AuthService } from '../../core/auth.service';
import { Cart } from '../../core/cart';
import { Usuario } from '../../core/auth';

@Component({
  selector: 'app-navbar',
  standalone: true,
  templateUrl: './navbar.html',
  styleUrls: ['./navbar.scss'],
  imports: [CommonModule, RouterModule],

/**
 * @description Barra de navegaci贸n principal. Muestra enlaces de la app,
 * el usuario autenticado (si existe) y la cantidad total de 铆tems en el carrito.
 * @usageNotes
 * - Se suscribe a `AuthService.usuarioActual$` para reaccionar a login/logout.
 * - Calcula la cantidad total del carrito escuchando `Cart.carrito$`.
 */
export class NavbarComponent implements OnInit, OnDestroy {
  usuario: Usuario | null = null;
  cantidadTotal = 0;

  // 猸锔 DECLARACIN CORREGIDA: Propiedad para gestionar la suscripci贸n de RxJS
  private userSubscription!: Subscription;

  /**
   * @description Inyecta los servicios de autenticaci贸n y carrito, y se suscribe
   * inmediatamente al carrito para mantener `cantidadTotal`.
   * @param auth Servicio de autenticaci贸n.
   * @param cart Servicio de carrito.
   */
  constructor(private auth: AuthService, private cart: Cart) {
    // Suscripci贸n al carrito (se mantiene)
    this.cart.carrito$.subscribe((items) => {
      this.cantidadTotal = items.reduce((acc, i) => acc + i.cantidad, 0);
    });
  }

  /**
   * @description Hook de inicializaci贸n. Se suscribe al observable
   * `usuarioActual$` para mantener `usuario` sincronizado con el estado
   * de sesi贸n.
   * @returns Nada (`void`).
   */
  // 1. Hook para inicializar y suscribirse al stream de usuario
  ngOnInit(): void {
    // Al suscribirnos a usuarioActual$, el componente recibe el usuario CADA VEZ que cambia (login/logout)
    this.userSubscription = this.auth.usuarioActual$.subscribe((user) => {
      this.usuario = user;
    });
  }

  /**
   * @description Hook de destrucci贸n. Libera la suscripci贸n a `usuarioActual$`
   * para evitar fugas de memoria.
   * @returns Nada (`void`).
   */
  // 2. Hook para limpiar la suscripci贸n al destruir el componente (buena pr谩ctica)
  ngOnDestroy(): void {
    if (this.userSubscription) {
      this.userSubscription.unsubscribe();
    }
  }

  /**
   * @description Cierra la sesi贸n del usuario actual delegando en `AuthService.logout`.
   * @returns Nada (`void`).
   */
  // M茅todo que llama a AuthService para cerrar la sesi贸n
  logout() {
    this.auth.logout();
  }
}
