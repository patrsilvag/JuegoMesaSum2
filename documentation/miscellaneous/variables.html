RoleGuard
Type : unknown
Default value : (roleRequerido: string) => { return () => { const auth = inject(AuthService); const router = inject(Router); const user = auth.getUsuarioActual(); if (!user) { router.navigate(['/login']); return false; } if (user.rol !== roleRequerido) { router.navigate(['/']); return false; } return true; }; }
